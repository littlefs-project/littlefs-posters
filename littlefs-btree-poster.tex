\block{Red-black-yellow Dhara (rbyd) trees}{\Large
    Extends Dhara trees with self-balancing operations:
    \begin{itemize}
        \item Based on red-black trees, but with slightly different coloring
            rules to avoid rotations.
        \item Instead of rotating, 2-3-4 splits point to existing branches in
            history.
        \item Unreachable branches are pruned during later appends.
    \end{itemize}
    Rbyd B-tree nodes support efficient sparse\slash order-statistic keys:
    \begin{itemize}
        %\item Order-statistic rbyd array insertion: $O(\log b)$
        \item Rbyd array insertion: $O(\log b)$

\block{Rbyd B-trees}{\Large
    \begin{itemize}
        \item Rbyd leaves point to B-tree nodes, inherting
            rbyd's sparse/order-statistic properties.
        \item Most updates are cheap $O(\log b)$ rbyd appends.
        %\item When logs are full, logs are compacted into new blocks.
        %    Compaction cost amortized $O(b \log b) \rightarrow O(\log b)$.
        \item When full, %logs
            rbyds are compacted into new blocks in $O(b \log b)$.
        \item To avoid runaway compactions, nodes split when $\gt b/2$,
            amortizing compaction cost $O(b \log b) \rightarrow O(\log b)$.
    \end{itemize}
}
\column{0.3}
\block{Rebalancing compactions}{\Large
    \begin{hackize}
        %Naively copying rbyds results in $O(b \log b)$ storage.
        \item Naively, rbyds result in $O(b \log b)$ storage.
        \item Instead, trees are copied with a tail-recursive rebalancing
            algorithm:
        \begin{itemize}
            %\item Brings compacted storage cost down
            %    $O(b \log b) \rightarrow O(b)$.
            \item Brings compacted storage cost down to $O(b)$.
            %\item Divides branching factor by a constant, currently {$\sim$}8.
            \item Divides branching factor by a constant ${\sim}1/8$.
        \end{itemize}
    \end{hackize}
}
\column{0.3}
\block{Deferred splits/merges}{\Large
    No reason to split/merge until compaction.
    \begin{itemize}
        \item Amortizes split/merge checks.
        % \item Introduces hysterisis, no split{$\leftrightarrow$}merge
        %     oscillation.
        \item Minimizes split{$\leftrightarrow$}merge oscillation.
        \item Results in optimal B-tree distribution after sequential
            writes.
        % \item Optimal B-tree distribution after sequential writes.
    \end{itemize}
}
