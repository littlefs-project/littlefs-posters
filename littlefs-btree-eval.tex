% x minor log ticks hack
\newcommand{\xminorlogticks}[1][tb]{
    % get jump/div
    \pgfkeysgetvalue{/pgfplots/xminorlogtick jump}{\xjump}
    \pgfkeysgetvalue{/pgfplots/xminorlogtick div}{\xdiv}
    % generate xminorlogticks
    \foreach[expand list=true] \x in {
            \pgfkeysvalueof{/pgfplots/xminorlogticks}}
        \foreach[parse=true] \i in {1,2,...,2^\xdiv-2^(\xdiv-\xjump)-1} {
            % tick locations
            \def\xi{(2^\x)*(1-\i/2^\xdiv)}
            % bottom ticks
            \IfSubStr{#1}{b}{
                \edef\expanded{\noexpand
                    \draw[style={thick,gray}]
                        ({{\xi},0})
                        -- ([yshift=+6pt]{{\xi},0});
                }
                \expanded
            }{}
            % top ticks
            \IfSubStr{#1}{t}{
                \edef\expanded{\noexpand
                    \draw[style={thick,gray}]
                        ({{\xi},0} |- current axis.north)
                        -- ([yshift=-6pt]{{\xi},0} |- current axis.north);
                }
                \expanded
            }{}
    }
}
